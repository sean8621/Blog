# 钉钉接口对接文档（Node.js）

## 一、获取 AccessToken（公共接口）

这个接口是 **所有钉钉开放接口的前置步骤**。

基本只要调用钉钉开放接口，都需要先获取 accessToken。

这个方法建议 **单独封装成公共方法**，以后调用钉钉接口都可以复用。

```javascript
const https = require("https");
const crypto = require("crypto");
const APP_KEY = process.env.DINGTALK_APP_KEY || "";
const APP_SECRET = process.env.DINGTALK_APP_SECRET || "";
const DINGTALK_API_HOST = "api.dingtalk.com";

function doRequest({ method, path, headers, body }) {
  return new Promise((resolve, reject) => {
    const options = {
      hostname: DINGTALK_API_HOST,
      port: 443,
      path,
      method,
      headers: headers || {},
      rejectUnauthorized: false,
    };

    const req = https.request(options, (res) => {
      let data = "";
      res.on("data", (chunk) => {
        data += chunk;
      });
      res.on("end", () => {
        try {
          const parsed = JSON.parse(data || "{}");
          if (res.statusCode >= 200 && res.statusCode < 300) {
            resolve(parsed);
          } else {
            reject(new Error(`HTTP ${res.statusCode}: ${data}`));
          }
        } catch (e) {
          reject(
            new Error(`Failed to parse response: ${e.message}, raw: ${data}`),
          );
        }
      });
    });

    req.on("error", (err) => {
      reject(err);
    });

    if (body) {
      req.write(body);
    }

    req.end();
  });
}

async function getAccessToken() {
  const path = "/v1.0/oauth2/accessToken";

  const body = JSON.stringify({
    appKey: APP_KEY,
    appSecret: APP_SECRET,
  });

  const res = await doRequest({
    method: "POST",
    path,
    headers: {
      "Content-Type": "application/json",
    },
    body,
  });

  if (!res || !res.accessToken) {
    throw new Error(`获取 accessToken 失败: ${JSON.stringify(res)}`);
  }

  return res.accessToken;
}
```

调用方式：

```javascript
const token = await getAccessToken();
```

## 二、IM机器人发送互动卡片

### 1 接入步骤

1. 企业内部应用创建机器人
2. 获取accessToken
3. 创建卡片模版
4. 调用创建并投放卡片接口

### 2 核心参数

发送卡片最重要的几个参数：

参数 说明

---

cardTemplateId 卡片模板ID
outTrackId 业务唯一ID
openSpaceId 卡片投放场域ID
cardData 卡片数据
imRobotOpenDeliverModel IM机器人单聊场域投放模型
imRobotOpenSpaceModel IM机器人单聊场域投放模型

### 3 JSON 示例

卡片内容通常来自业务数据，例如审批信息。根据卡片模板，**必须把所有字段都填上**。

```javascript
// 员工的 userId 列表
const RECEIVER_USER_IDS = [""];

// 互动卡片模板 ID（在钉钉管理后台 - 互动卡片 builder 创建后获取）
const CARD_TEMPLATE_ID = "";
const cardDataCardParamMap = {
  title: "",
  salePlanName: "",
  creator: "",
  createTime: "",
  currentNode: "",
  jump_url: "",
  status: "",
  color: "",
};

// 生成唯一的 outTrackId（外部业务编码），示例使用时间戳 + 随机串
function generateOutTrackId() {
  const ts = Date.now();
  const rand = crypto.randomBytes(4).toString("hex");
  return `card_${ts}_${rand}`;
}
// 由于接口要求卡片数据的键值对均为 string 类型，需要对卡片数据进行预处理
function convertJSONValuesToString(obj) {
  const result = {};
  for (const [key, value] of Object.entries(obj)) {
    if (typeof value === "string") {
      result[key] = value;
    } else {
      try {
        result[key] = JSON.stringify(value);
      } catch (e) {
        result[key] = "";
      }
    }
  }
  return result;
}
async function sendInteractiveCard(accessToken) {
  const path = "/v1.0/card/instances/createAndDeliver";
  const outTrackId = generateOutTrackId();
  const cardData = {
    cardParamMap: convertJSONValuesToString(cardDataCardParamMap),
  };

  const payload = {
    cardTemplateId: CARD_TEMPLATE_ID,
    outTrackId,
    openSpaceId: `dtv1.card//IM_ROBOT.${RECEIVER_USER_IDS[0]}`,
    imRobotOpenSpaceModel: {
      supportForward: false,
      lastMessageI18n: { ZH_CN: "机器人单聊通知" },
      searchSupport: {
        searchIcon: "",
        searchTypeName: '{"zh_CN":"机器人单聊通知"}',
        searchDesc: "机器人单聊通知",
      },
      notification: {
        alertContent: "您有新的机器人单聊通知",
        isNotificationOff: false,
      },
    },
    imRobotOpenDeliverModel: {
      spaceType: "IM_ROBOT",
      robotCode: APP_KEY,
    },
    cardData,
  };

  const body = JSON.stringify(payload);

  console.log("body", body);

  const res = await doRequest({
    method: "POST",
    path,
    headers: {
      "Content-Type": "application/json",
      "x-acs-dingtalk-access-token": accessToken,
    },
    body,
  });

  return { res, outTrackId };
}
```

### 4 接入过程中遇到的坑

#### 4.1 每个场域的 OpenSpaceModel 和 OpenDeliverModel 必须配置

这是调试时间 **最长的一个坑**。

钉钉互动卡片支持多种投放场域，例如：

- 机器人单聊
- 群聊
- 工作通知
- 应用空间

不同场域需要不同的 `OpenSpaceModel` 和 `OpenDeliverModel`。

如果不配置或者配置错误，就会报错。

例如机器人单聊场景：

```json
{
  "imRobotOpenSpaceModel": {},
  "imRobotOpenDeliverModel": {}
}
```

示例代码：

```javascript
imRobotOpenSpaceModel: {
  supportForward: false,
  lastMessageI18n: {
    "ZH_CN": "销量计划通知"
  }
},

imRobotOpenDeliverModel: {
  spaceType: "IM_ROBOT",
  robotCode: APP_KEY
}
```

如果缺少这些字段，可能出现以下报错：

- invalid parameter
- deliver model not match
- openSpaceId invalid

而且官方文档通常不会明确告诉你具体缺哪个字段，只能对照示例慢慢调试。

#### 4.2 openSpaceId 格式

机器人单聊通常是：

dtv1.card//IM_ROBOT.userId

例如：

dtv1.card//IM_ROBOT.123456
